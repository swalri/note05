<script type="text/javascript">
const images = ["images/최근상품01.jpg","images/최근상품02.jpg",
"images/최근상품03.jpg","images/최근상품04.jpg","images/최근상품05.jpg",
"images/최근상품06.jpg","images/최근상품07.jpg","images/최근상품08.jpg"];
window.onload=function init(){
console.log("init")
let msg = "<div class='flex'>";
for(let i=0; i < images.length ; i++){
msg += `<div class="img" id="${i}" onMouseover="imgOver(${i})"
onMouseout="imgOut(${i})">`;
msg += `<img src="${images[i]}" >`
msg += "</div>"
if(i+1 == Number(images.length/2))
msg +="</div><div class='flex'>"
}
const wrap = document.getElementById("wrap");
wrap.innerHTML = msg;
}
const imgOver=(index)=>{
const span = document.getElementById(index);
span.innerHTML = "<span>선택</span>";
}
const imgOut=(index)=>{
const div = document.getElementById(index);
div.innerHTML = `<img src="${images[index]}">`;
}
